import numpy as np
from scipy.spatial.distance import cdist
from scipy.stats import spearmanr, pearsonr
from sklearn.linear_model import LogisticRegression
from sklearn.model_selection import cross_val_predict
from sklearn.pipeline import make_pipeline
from sklearn.preprocessing import StandardScaler
from sklearn.metrics import roc_auc_score

# =========================
# LOAD DATA
# =========================
X = np.load("brain_11d.npy")
y = np.load("labels.npy")

labels = np.unique(y)

# =========================
# CENTROIDS
# =========================
centroids = np.array([X[y == l].mean(axis=0) for l in labels])

# =========================
# DISTANCES + MARGIN
# =========================
dists = cdist(X, centroids)

d_correct = []
d_margin = []

for i in range(len(X)):
    true_idx = np.where(labels == y[i])[0][0]
    
    dc = dists[i, true_idx]
    d_other = np.min(np.delete(dists[i], true_idx))
    
    d_correct.append(dc)
    d_margin.append(dc - d_other)

d_correct = np.array(d_correct)
d_margin = np.array(d_margin)

# =========================
# CLASSIFIER UNCERTAINTY
# =========================
model = make_pipeline(
    StandardScaler(),
    LogisticRegression(max_iter=5000)
)

probs = cross_val_predict(model, X, y, cv=5, method="predict_proba")

p_correct = []

for i in range(len(y)):
    true_idx = np.where(labels == y[i])[0][0]
    p_correct.append(probs[i, true_idx])

p_correct = np.array(p_correct)
uncertainty = 1 - p_correct

# =========================
# CORRELATIONS
# =========================
r_margin, p_margin = spearmanr(d_margin, uncertainty)
r_dist, p_dist = spearmanr(d_correct, uncertainty)

r_margin_p, _ = pearsonr(d_margin, uncertainty)
r_dist_p, _ = pearsonr(d_correct, uncertainty)

# =========================
# AUC TEST
# =========================
high_uncertainty = uncertainty > np.median(uncertainty)

auc_margin = roc_auc_score(high_uncertainty, d_margin)
auc_dist = roc_auc_score(high_uncertainty, d_correct)

# =========================
# SHUFFLE CONTROL
# =========================
rng = np.random.default_rng(42)
X_shuff = X.copy()

for col in range(X.shape[1]):
    rng.shuffle(X_shuff[:, col])

centroids_shuff = np.array([X_shuff[y == l].mean(axis=0) for l in labels])
dists_shuff = cdist(X_shuff, centroids_shuff)

d_margin_shuff = []

for i in range(len(X)):
    true_idx = np.where(labels == y[i])[0][0]
    
    dc = dists_shuff[i, true_idx]
    d_other = np.min(np.delete(dists_shuff[i], true_idx))
    
    d_margin_shuff.append(dc - d_other)

d_margin_shuff = np.array(d_margin_shuff)

r_shuff, p_shuff = spearmanr(d_margin_shuff, uncertainty)

# =========================
# PRINT RESULTS
# =========================
print("\n==============================")
print(" GEOMETRY → UNCERTAINTY TEST ")
print("==============================\n")

print("---- CORRELATIONS ----")
print(f"Margin vs Uncertainty (Spearman): r = {r_margin:.4f}, p = {p_margin:.2e}")
print(f"Distance vs Uncertainty (Spearman): r = {r_dist:.4f}, p = {p_dist:.2e}")

print(f"Margin vs Uncertainty (Pearson): r = {r_margin_p:.4f}")
print(f"Distance vs Uncertainty (Pearson): r = {r_dist_p:.4f}")

print("\n---- PREDICTIVE POWER (AUC) ----")
print(f"AUC (Margin): {auc_margin:.4f}")
print(f"AUC (Distance): {auc_dist:.4f}")

print("\n---- SHUFFLE CONTROL ----")
print(f"Shuffled Margin vs Uncertainty: r = {r_shuff:.4f}, p = {p_shuff:.2e}")

print("\n---- INTERPRETATION ----")
if auc_margin > 0.75:
    print("✔ Strong evidence: ambiguity is driven by centroid competition (margin)")
else:
    print("⚠ Weak margin signal")

if auc_dist < 0.6:
    print("✔ Distance alone is NOT a good predictor of ambiguity")
else:
    print("⚠ Distance may carry signal")

if abs(r_shuff) < 0.05:
    print("✔ Structure disappears under shuffle → NOT random")
else:
    print("⚠ Potential random structure")

print("\n==============================\n")
